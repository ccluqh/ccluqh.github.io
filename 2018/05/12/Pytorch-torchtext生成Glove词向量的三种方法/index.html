Pytorch torchtext生成Glove词向量的三种方法

May 12, 2018点击

import collections
from torchtext.vocab import Vocab
from torchtext.vocab import GloVe
from torchtext import data, datasets

##方法一：
TEXT = data.Field(lower=True)
LABEL = data.Field(sequential=False)

train, test = datasets.IMDB.splits(TEXT, LABEL)

TEXT.build_vocab(train, vectors=GloVe(name=’6B’, dim=100))
LABEL.build_vocab(train,)

#torch.Size([251639, 100])
print(TEXT.vocab.vectors.size())

print(TEXT.vocab.stoi.get(“cheonhye”))
print(TEXT.vocab.vectors[TEXT.vocab.stoi.get(“cheonhye”)])
print(TEXT.vocab.vectors[TEXT.vocab.stoi.get(“cheonhye”)].size())

print(TEXT.vocab.stoi.get(“hello”))
print(“hello:”,TEXT.vocab.vectors[TEXT.vocab.stoi.get(“hello”)])
print(TEXT.vocab.vectors[TEXT.vocab.stoi.get(“hello”)].size())

方法二：

vectors = GloVe(name=’6B’, dim=’100’)

#建立测试集词典
c = collections.Counter({‘hello’: 10, ‘world’: 7, ‘ᑌᑎIᑕOᗪᕮ_Tᕮ᙭T’: 5, ‘freq_too_low’: 2})
v = Vocab(c, min_freq=1, specials=[‘\<unk>‘, ‘\<pad>‘, ‘\<bos>‘], vectors=vectors)

#测试集词典的向量表示
print(“v.vectors:”, v.vectors)

for w, vec in zip(v.stoi.keys(), v.vectors):

print(w, vec[:5].view(1, -1))

print(v.stoi.get(“cheonhye”))
print(v.vectors[v.stoi.get(“cheonhye”)])
print(v.vectors[v.stoi.get(“cheonhye”)].size())

方法三:

glove = GloVe(name=’6B’, dim=100)
print(“hello:”, glove.vectors[glove.stoi[“hello”]])
